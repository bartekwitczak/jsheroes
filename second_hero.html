Functions

Defining


function welcome(name) {
    console.log("Welcome" + name);
}

var factorial = function(n) {
    if(n <= 1) return 1;
    return n * factorial(n - 1);
};

Function params


var separate = function(array, separator) {
    return array.join(separator);
};

separate([1, 2, 3, 4], "-");

Function params - optional


var separate = function(array, separator) {
    if(separator === undefined) separator = "-";
    return array.join(separator);
};

separate([1, 2, 3, 4], "-");
separate([1, 2, 3, 4]);

Function params - arguments


var addElements = function(array) {
    var copy = [];
    array.forEach(function(elem) { copy.push(elem); });
    for(var i = 1; i < arguments.length; i++)
        copy.push(arguments[i]);

    return copy;
};

addElements([]);
addElements([], 1, 2, 3, 4);
